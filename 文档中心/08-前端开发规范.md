## 三、路由规范

### 3.1 路由定义

```javascript
// js/app.js
const routes = [
    { path: '/', component: NewsView },
    { path: '/skills', component: SkillsView },
    { path: '/skill/:id', component: SkillDetailView },
    { path: '/subscribe', component: SubscribeView },
    {
        path: '/sys',
        component: AdminLayout,
        children: [
            { path: '', redirect: '/sys/news' },
            { path: 'news', component: NewsAdmin },
            { path: 'skills', component: SkillsAdmin },
            { path: 'subscribe', component: SubscribeAdmin },
            { path: 'versions', component: VersionsAdmin },
            { path: 'docs', component: DocsAdmin }
        ]
    }
];
```

### 3.2 路由规范

- 公开页面：`/`, `/skills`, `/skill/:id`, `/subscribe`
- 管理后台：`/sys` 开头，使用AdminLayout作为父组件
- 动态参数：使用 `:param` 语法

## 五、API请求规范

### 5.1 API封装

```javascript
// js/api.js
const API = {
    async get(url) {
        const res = await fetch(url);
        return res.json();
    },
    async post(url, data) {
        const res = await fetch(url, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(data)
        });
        return res.json();
    },
    async put(url, data) {
        const res = await fetch(url, {
            method: 'PUT',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(data)
        });
        return res.json();
    },
    async del(url) {
        const res = await fetch(url, { method: 'DELETE' });
        return res.json();
    }
};
```

### 5.2 调用示例

```javascript
// GET请求
const news = await API.get('/api/news/search?keyword=AI&page=1');

// POST请求
const result = await API.post('/api/skills', {
    name: 'New Skill',
    category: 'AI'
});

// PUT请求
await API.put('/api/skills/1', { name: 'Updated Name' });

// DELETE请求
await API.del('/api/skills/1');
```

### 5.3 错误处理

```javascript
async function safeRequest(fn) {
    try {
        const result = await fn();
        if (result.code !== 200) {
            alert(result.message || '操作失败');
            return null;
        }
        return result;
    } catch (error) {
        console.error('请求错误:', error);
        alert('网络错误，请稍后重试');
        return null;
    }
}

// 使用
const data = await safeRequest(() => API.get('/api/news/search'));
```

## 六、页面开发规范

### 6.1 页面组件结构

```javascript
const NewsView = {
    data() {
        return {
            newsList: [],
            keyword: '',
            loading: false,
            pagination: { page: 1, total: 0 }
        };
    },
    methods: {
        async search() {
            this.loading = true;
            try {
                const url = `/api/news/search?keyword=${encodeURIComponent(this.keyword)}&page=${this.pagination.page}`;
                const res = await API.get(url);
                if (res.code === 200) {
                    this.newsList = res.data || [];
                    this.pagination.total = res.total || 0;
                }
            } finally {
                this.loading = false;
            }
        },
        onPageChange(page) {
            this.pagination.page = page;
            this.search();
        }
    },
    mounted() {
        this.search();
    },
    template: `
        <div class="news-view">
            <input v-model="keyword" @keyup.enter="search" placeholder="搜索新闻..." />
            <button @click="search">搜索</button>
            <div v-if="loading">加载中...</div>
            <div v-else class="news-list">
                <div v-for="news in newsList" :key="news.id" class="news-item">
                    <h3>{{ news.title }}</h3>
                    <p>{{ news.summary }}</p>
                </div>
            </div>
        </div>
    `
};
```

### 6.2 列表页规范

- 搜索/筛选区域在上方
- 列表主体在中间
- 分页组件在底部
- loading状态要有明确反馈
